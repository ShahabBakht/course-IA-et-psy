2. Comprendre la fonction sigmo√Øde#

La fonction sigmo√Øde est au c≈ìur de la r√©gression logistique. Elle transforme n‚Äôimporte quelle valeur r√©elle en une valeur entre 0 et 1.

\[\sigma(z) = \frac{1}{1 + e^{-z}}\]
def sigmoid(z):
    """
    Fonction sigmo√Øde.
    
    Parameters:
    -----------
    z : float or array-like
        Valeur(s) d'entr√©e
    
    Returns:
    --------
    float or array-like
        Valeur(s) entre 0 et 1
    """
    return 1 / (1 + np.exp(-z))

4. Impl√©mentation from scratch : R√©gression logistique#

Impl√©mentons la r√©gression logistique avec la descente de gradient.

Rappel :

  • Mod√®le : \(\hat{y} = \sigma(Wx + b)\)

  • Loss : \(L = -\frac{1}{n} \sum [y \log(\hat{y}) + (1-y) \log(1-\hat{y})]\)

  • Gradients :

    • \(\frac{\partial L}{\partial w} = \frac{1}{n} X^T(\hat{y} - y)\)

    • \(\frac{\partial L}{\partial b} = \frac{1}{n} \sum (\hat{y} - y)\)

class LogisticRegressionGD:
    """
    R√©gression logistique avec descente de gradient.
    """
    def __init__(self, learning_rate=0.01, n_iterations=1000):
        self.lr = learning_rate
        self.n_iterations = n_iterations
        self.w = None
        self.b = None
        self.loss_history = []
    
    def sigmoid(self, z):
        """Fonction sigmo√Øde."""
        return 1 / (1 + np.exp(-z))
    
    def compute_loss(self, y, y_pred):
        """
        Calculer la binary cross-entropy loss.
        
        Ajoute un petit epsilon pour √©viter log(0).
        """
        epsilon = 1e-15
        y_pred = np.clip(y_pred, epsilon, 1 - epsilon)
        loss = -np.mean(y * np.log(y_pred) + (1 - y) * np.log(1 - y_pred))
        return loss
    
    def fit(self, X, y):
        """
        Entra√Æner le mod√®le.
        
        Parameters:
        -----------
        X : array-like, shape (n_samples, n_features)
        y : array-like, shape (n_samples,)
        """
        n_samples, n_features = X.shape
        
        # Initialisation
        self.w = np.zeros((n_features, 1))
        self.b = 0
        
        # Reshape y
        y = y.reshape(-1, 1)
        
        # Descente de gradient
        for i in range(self.n_iterations):
            # Pr√©diction
            z = X.dot(self.w) + self.b
            y_pred = self.sigmoid(z)
            
            # Calcul de la loss
            loss = self.compute_loss(y, y_pred)
            self.loss_history.append(loss)
            
            # Calcul des gradients
            dw = (1 / n_samples) * X.T.dot(y_pred - y)
            db = (1 / n_samples) * np.sum(y_pred - y)
            
            # Mise √† jour des param√®tres
            self.w -= self.lr * dw
            self.b -= self.lr * db
            
            # Affichage occasionnel
            if (i + 1) % 100 == 0:
                print(f"It√©ration {i+1}/{self.n_iterations}, Loss: {loss:.4f}")
    
    def predict_proba(self, X):
        """
        Pr√©dire les probabilit√©s.
        """
        z = X.dot(self.w) + self.b
        return self.sigmoid(z)
    
    def predict(self, X, threshold=0.5):
        """
        Pr√©dire les classes.
        """
        y_pred_proba = self.predict_proba(X)
        return (y_pred_proba >= threshold).astype(int)
    
    def get_params(self):
        """
        Retourner les param√®tres.
        """
        return {'w': self.w, 'b': self.b}

Visualisation de la fronti√®re de d√©cision#

La fronti√®re de d√©cision est l‚Äôensemble des points o√π \(\hat{y} = 0.5\), c‚Äôest-√†-dire o√π \(w_1 x_1 + w_2 x_2 + b = 0\).

def plot_decision_boundary(model, X, y, title="Fronti√®re de d√©cision"):
    """
    Visualiser la fronti√®re de d√©cision.
    """
    # Cr√©er une grille de points
    x1_min, x1_max = X[:, 0].min() - 1, X[:, 0].max() + 1
    x2_min, x2_max = X[:, 1].min() - 1, X[:, 1].max() + 1
    xx1, xx2 = np.meshgrid(np.linspace(x1_min, x1_max, 100),
                           np.linspace(x2_min, x2_max, 100))
    
    # Pr√©dire sur tous les points de la grille
    X_grid = np.c_[xx1.ravel(), xx2.ravel()]
    Z = model.predict(X_grid)
    Z = Z.reshape(xx1.shape)
    
    # Visualisation
    plt.figure(figsize=(10, 6))
    plt.contourf(xx1, xx2, Z, alpha=0.3, levels=1, colors=['blue', 'red'])
    plt.contour(xx1, xx2, Z, levels=[0.5], colors='black', linewidths=2)
    
    # Points de donn√©es
    plt.scatter(X[y==0, 0], X[y==0, 1], c='blue', label='Classe 0', 
                alpha=0.7, edgecolors='k', s=50)
    plt.scatter(X[y==1, 0], X[y==1, 1], c='red', label='Classe 1', 
                alpha=0.7, edgecolors='k', s=50)
    
    plt.xlabel('Feature 1', fontsize=12)
    plt.ylabel('Feature 2', fontsize=12)
    plt.title(title, fontsize=14)
    plt.legend(fontsize=11)
    plt.grid(True, alpha=0.3)
    plt.show()

# Visualiser
plot_decision_boundary(model_scratch, X, y,

# Simuler des donn√©es multi-classes
np.random.seed(42)

n_trials = 400  # Nombre d'essais
n_neurons = 20  # Nombre de neurones
n_classes = 4   # 4 directions: droite, gauche, haut, bas

# G√©n√©rer les donn√©es
X_multi = np.random.randn(n_trials, n_neurons) * 10 + 30  # Firing rates

# Ajouter des patterns pour chaque classe
for i in range(n_classes):

    # Chaque classe a un pattern distinct
